      set user and group in docker compose

      2021-09-11 docker

      Step 1: Add user entry in docker-compose.yml

      version: '3'
services:
    app:
        image: alpine
        user: "${UID}:${GID}"

      step 2: run docker-compose

      env UID=${UID} GID=${GID} docker-compose run app id
